## prisma/schema.prisma (Complete)

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

model User {
  id        String   @id @default(cuid())
  email     String   @unique
  password  String   // bcryptjs hashed
  fullName  String?
  createdAt DateTime @default(now())

  applications JobApplication[]
  notes        Note[]
  sessions     Session[]
}

model Session {
  id        String   @id @default(cuid())
  userId    String
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  token     String   @unique
  expiresAt DateTime
  createdAt DateTime @default(now())
}

model JobApplication {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  company     String
  role        String
  jobUrl      String?
  location    String?
  salaryMin   Int?
  salaryMax   Int?
  stage       Stage    @default(APPLIED)
  priority    Priority @default(MEDIUM)
  source      String?
  appliedDate DateTime

  contacts    Json     @default("[]")   // Array of Contact objects
  documents   Json     @default("{}")   // Object with resume/cover letter URLs

  deletedAt   DateTime?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  notes       Note[]
  events      ApplicationEvent[]

  @@index([userId])
  @@index([stage])
  @@fulltext([company, role])  // For full-text search (SQLite FTS5)
}

model Note {
  id            String   @id @default(cuid())
  applicationId String
  application   JobApplication @relation(fields: [applicationId], references: [id], onDelete: Cascade)
  userId        String
  user          User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  content       String   @db.Text
  createdAt     DateTime @default(now())

  @@index([applicationId])
  @@index([userId])
}

model ApplicationEvent {
  id            String   @id @default(cuid())
  applicationId String
  application   JobApplication @relation(fields: [applicationId], references: [id], onDelete: Cascade)

  eventType     EventType
  oldValue      String?
  newValue      String?
  createdAt     DateTime @default(now())

  @@index([applicationId])
}

enum Stage {
  APPLIED
  PHONE_SCREEN
  INTERVIEW
  OFFER
  REJECTED
}

enum Priority {
  LOW
  MEDIUM
  HIGH
}

enum EventType {
  CREATED
  STAGE_CHANGED
  NOTE_ADDED
  NOTE_DELETED
}
```

## lib/prisma.ts (Singleton Client)

```typescript
// lib/prisma.ts
import { PrismaClient } from '@prisma/client';

const globalForPrisma = global as unknown as { prisma: PrismaClient };

export const prisma =
  globalForPrisma.prisma ||
  new PrismaClient({
    log: process.env.NODE_ENV === 'development' ? ['query'] : []
  });

if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma;
```

```typescript
// prisma/seed.ts
import { prisma } from '../lib/prisma';
import * as bcrypt from 'bcryptjs';

async function main() {
  // Clear existing data
  await prisma.applicationEvent.deleteMany();
  await prisma.note.deleteMany();
  await prisma.jobApplication.deleteMany();
  await prisma.session.deleteMany();
  await prisma.user.deleteMany();

  // Create test user
  const hashedPassword = await bcrypt.hash('password123', 10);
  const user = await prisma.user.create({
    data: {
      email: 'test@example.com',
      password: hashedPassword,
      fullName: 'Test User'
    }
  });

  // Create sample applications
  const apps = await Promise.all([
    prisma.jobApplication.create({
      data: {
        userId: user.id,
        company: 'Google',
        role: 'Senior Software Engineer',
        jobUrl: 'https://google.com/jobs/123',
        location: 'Mountain View, CA',
        stage: 'PHONE_SCREEN',
        priority: 'HIGH',
        source: 'LinkedIn',
        appliedDate: new Date('2025-01-15'),
        salaryMin: 150000,
        salaryMax: 200000
      }
    }),
    prisma.jobApplication.create({
      data: {
        userId: user.id,
        company: 'Meta',
        role: 'Software Engineer',
        jobUrl: 'https://meta.com/jobs/456',
        location: 'Menlo Park, CA',
        stage: 'INTERVIEW',
        priority: 'MEDIUM',
        source: 'Company Website',
        appliedDate: new Date('2025-01-10')
      }
    }),
    prisma.jobApplication.create({
      data: {
        userId: user.id,
        company: 'Apple',
        role: 'iOS Developer',
        location: 'Cupertino, CA',
        stage: 'APPLIED',
        priority: 'LOW',
        source: 'Referral',
        appliedDate: new Date('2025-01-20')
      }
    })
  ]);

  // Create sample notes
  await prisma.note.create({
    data: {
      applicationId: apps[0].id,
      userId: user.id,
      content: 'Great conversation with recruiter. Strong culture fit. Follow-up call scheduled.'
    }
  });

  // Create events
  await Promise.all([
    prisma.applicationEvent.create({
      data: {
        applicationId: apps[0].id,
        eventType: 'CREATED',
        newValue: 'APPLIED'
      }
    }),
    prisma.applicationEvent.create({
      data: {
        applicationId: apps[0].id,
        eventType: 'STAGE_CHANGED',
        oldValue: 'APPLIED',
        newValue: 'PHONE_SCREEN'
      }
    })
  ]);

  console.log('✅ Database seeded with test data');
}

main()
  .catch((e) => {
    console.error(e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
```

## Implementation Steps

1. Copy `prisma/schema.prisma` to project
2. Run `npx prisma migrate dev --name init` (creates `prisma/dev.db`)
3. Create `lib/prisma.ts` singleton
4. Create `types/index.ts` with domain types
5. (Optional) Create `prisma/seed.ts` and run `npm run db:seed`
6. Verify database created: `ls -la prisma/dev.db`
